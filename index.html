New ECMAScript features
var, let, const 

					// ES5
					var a = 5; // global scope, function scope
	
					// ES6
					let a = 5 // block scope
example: var 

					var elements = ['A', 'B', 'C'];
	
					for (var i = 0; i < elements.length; i++) {
					  console.log('element: ', elements[i]);
					}
	
					console.log(i); // will print 3
example: let 

					var elements = ['A', 'B', 'C'];
	
					for (let i = 0; i < elements.length; i++) {
					  console.log('element: ', elements[i]);
					}
	
					console.log(i); // Uncaught ReferenceError: i is not defined
example: let vs. const 

				let a = 5;
				a = 3;

				console.log(a); // will print 3

				
				const a = 5;
				a = 3; // Uncaught TypeError: Assignment to constant variable.
example: let vs. const 

				const a = [1, 2, 3];
				a.push(4); // OK, it's not reassignment

				const a = { a: 'A', b: 'B'};
				a['c'] = 'C'; // OK
arrow functions
ES 5

              var a = function() {
                return 'a';
              }

              function a() {
                return 'a';
              }
ES 6

              const a = () => {
                return 'a';
              }

              const a = () => 'a';
Lexical this

              // ES5
              var a = (function() {
                return 'a';
              }).bind(this);              
            

              // ES6
              const a = () => {
                return 'a';
              }
functions - parameter handling
default argument value 

              // ES5
              var func = function(a) {
                if (typeof a === 'undefined') {
                  a = 2;
                }

                return a;
              }
            

              // ES6
              const func = (a = 2) => {
                return a;
              }

              // shorthand
              const func = (a = 2) => a;
rest parameter 

              // ES5
              function func (a, b) {
                var rest = Array.prototype.slice.call(arguments, 2);
                
                return a + ',' + b + ',' + rest.join(',');
              };
            

              // ES6
              const func = (a, b, ...rest) => a + ',' + b + ',' + rest.join(',');
spread operator 

              // ES5
              var params = [4, 5, 6];

              [1, 2, 3].concat(params2); // [1, 2, 3, 4, 5, 6]
            

              // ES6
              const params = [4, 5, 6];

              [1, 2, 3, ...params2]; // [1, 2, 3, 4, 5, 6]
