New ECMAScript features

var, let, const

var vs. let


					// ES5
					var a = 5; // global scope, function scope
	
					// ES6
					let a = 5 // block scope

example - var


					var elements = ['A', 'B', 'C'];
	
					for (var i = 0; i < elements.length; i++) {
					  console.log('element: ', elements[i]);
					}
	
					console.log(i); // will print 3

example - let


					var elements = ['A', 'B', 'C'];
	
					for (let i = 0; i < elements.length; i++) {
					  console.log('element: ', elements[i]);
					}
	
					console.log(i); // Uncaught ReferenceError: i is not defined

example - let vs. const


				let a = 5;
				a = 3;

				console.log(a); // will print 3
				
				const a = 5;
				a = 3; // Uncaught TypeError: Assignment to constant variable.

example - let vs. const


				const a = [1, 2, 3];
				a.push(4); // OK, it's not reassignment

				const a = { a: 'A', b: 'B'};
				a['c'] = 'C'; // OK

arrow functions

ES 5


              var a = function() {
                return 'a';
              }

              function a() {
                return 'a';
              }

ES 6


              const a = () => {
                return 'a';
              }

              const a = () => 'a';

Lexical this

ES 5

              var a = (function() {
                return 'a';
              }).bind(this);
ES 6

              const a = () => {
                return 'a';
              }

functions - parameter handling

default argument value

ES 5

              var func = function(a) {
                if (typeof a === 'undefined') {
                  a = 2;
                }

                return a;
              }
ES 6

              const func = (a = 2) => {
                return a;
              }

              // shorthand
              const func = (a = 2) => a;

rest parameter

ES 5

              function func (a, b) {
                var rest = Array.prototype.slice.call(arguments, 2);
                
                return a + ',' + b + ',' + rest.join(',');
              };
ES 6

              const func = (a, b, ...rest) => {
                return a + ',' + b + ',' + rest.join(',');
              }
            

              const func = (a, b, ...rest) => a + ',' + b + ',' + rest.join(',');

spread operator

ES 5

			  var params = [4, 5, 6];

			  [1, 2, 3].concat(params2); // [1,2,3,4,5,6]
ES 6

              // ES6
              const params = [4, 5, 6];

              [1, 2, 3, ...params2]; // [1,2,3,4,5,6]

Destructuring

array matching

example 1

          const list = [ 1, 2, 3 ];
          const [ a, b, c ] = list;
example 2

          const list = [ 1, 2, 3 ];
          const [ a,,b ] = list;

object Matching

example

          const object = {
            name: 'Konrad',
            age: 18,
            hobby: 'drums'
          };

          const { name, age, hobby } = object;
example - default values

          const object = {
            name: 'Konrad',
            hobby: 'drums'
          };

          const { name, age = 18, hobby } = object;

object matching - function arguments

example 1

          const func = (props) {
            const { name, value } = props;
            console.log(name, value);
          }
example 2

          const func = ({ name, value }) {
            console.log(name, value);
          }
equivalent to:

          const func = (props) {
            console.log(props.name, props.value);
          }

template strings

ES 5

          function func(a, b) {
            var rest = Array.prototype.slice.call(arguments, 2);
            
            return a + ',' + b + ',' + rest.join(',');
          };
ES 6

          const func = (a, b, ...rest) => {
            return `${a}, ${b}, ${rest.join(',')}`;
          }

object property shorthand

ES 5

        var x = 0, y = 0;
        obj = { x: x, y: y };
ES 6

        const x = 0, y = 0;
        obj = { x, y };

exports

export (named)

example 1

            export const App = ({ children }) => {
              return <div>{children}</div>
            }
example 2

            const App = ({ children }) => {
              return <div>{children}</div>
            }

            export { App };

import (named)


            // exactly name must be specified

            import { App } from './file.js';

export (unnamed)

example 1

            const App = ({ children }) => {
              return <div>{children}</div>
            }

            export default App;
example 2

            export default ({ children }) => {
              return <div>{children}</div>
            }

import (unnamed)


            /* name can be any */

            import App from './file.js';

            import AnotherAppName from './file.js';

classes

example

            class Shape {
              constructor(id, x, y) {
                  this.id = id
                  this.move(x, y)
              }

              move(x, y) {
                  this.x = x
                  this.y = y
              }
            }

async / await

example

            const fetchData = () => {
              /* asynchronous function */
            }

            const msg = async () => {
              const data = await fetchData();

              console.log('data:', data);
            }
example - with error handling

            const fetchData = () => {
              /* asynchronous function */
            }

            const msg = async () => {
              try {
                const data = await fetchData();

                console.log('data:', data);
              } catch(err) {
                throw new Error(err);
              }
            }
